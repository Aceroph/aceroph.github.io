	next
			-- Move to next index
		require
			has_more_elements: has_next
		do
			index := index + 1
		ensure
			incremented: old index + 1 = index
		end

	previous
			-- Move to previous index
		require
			not_is_first: has_previous
		do
			index := index - 1
		ensure
			incremented: old index - 1 = index
		end

	read: CHARACTER_8
			-- Read character
		do
			if not representation.is_empty then
				Result := representation.item (index)
			end
		end

	reset
			-- Reset reader
		do
			index := 1
		end

	set_representation (a_json: STRING_8)
			-- Set representation.
		do
			a_json.left_adjust
			a_json.right_adjust
			representation := a_json
			reset
		end

	skip_white_spaces
			-- Remove white spaces
		local
			c: like actual
		do
			from
				c := actual
			until
				(c /= ' ' and c /= '%N' and c /= '%R' and c /= '%U' and c /= '%T') or not has_next
			loop
				next
				c := actual
			end
		end
