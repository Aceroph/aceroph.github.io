deferred class 
	HIERARCHICAL [G]

inherit
	TRAVERSABLE [G]

feature -- Access

	successor_count: INTEGER_32
			-- Number of successors of current element
		require
			not_off: not off
		deferred
		end
	
feature -- Cursor movement

	up
			-- Move to predecessor.
		require
			not_off: not off
		deferred
		end

	down (i: INTEGER_32)
			-- Move to i-th successor.
		require
			not_off: not off
			argument_within_bounds: i >= 1 and i <= successor_count
		deferred
		end
	
invariant
	non_negative_successor_count: successor_count >= 0
