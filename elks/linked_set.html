class 
	LINKED_SET [G]

inherit
	LINEAR_SUBSET [G]
		undefine
			prune_all,
			copy,
			is_equal,
			fill
		select
			extend,
			put,
			prune
		end

	LINKED_LIST [G]
		rename
			extend as ll_extend,
			put as ll_put,
			prune as ll_prune
		export
			{NONE} ll_extend, ll_put, ll_prune
		undefine

			is_inserted
		redefine
			make_from_iterable
		end

create 
	make,
	make_from_iterable

feature {NONE} -- Creation

	make_from_iterable (other: ITERABLE [G])
			-- Create a set with all items obtained from other.
		do
			make
			across
				other as o
			loop
				extend (o.item)
			end
		end
	
feature -- Element change

	put (v: G)
			-- Ensure that set includes v.
			-- Was declared in LINKED_SET as synonym of extend.
		do
			if is_empty or else not has (v) then
				ll_extend (v)
			end
		end

	extend (v: G)
			-- Ensure that set includes v.
			-- Was declared in LINKED_SET as synonym of put.
		do
			if is_empty or else not has (v) then
				ll_extend (v)
			end
		end
	
feature -- Removal

	prune (v: like item)
			-- Remove v if present.
		do
			start
			ll_prune (v)
		end
