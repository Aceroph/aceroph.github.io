	make (a_physical_code, a_virtual_code: NATURAL_32; a_modifier: NATURAL_16; a_repeat: NATURAL_8)

		do
			physical_code := a_physical_code
			virtual_code := a_virtual_code
			repeat := a_repeat
			modifier := a_modifier
		end

		do
			Result := repeat /= 0
		end

	repeat: NATURAL_8

	modifier: NATURAL_16
			-- The internal identificator of the modifiers pressed
