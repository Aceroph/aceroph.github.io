class 
	GAME_COLOR_PALETTE

inherit

		redefine
			is_equal
		end

			make as make_structure,

create {GAME_PIXEL_FORMAT_READABLE}

create

feature {NONE} -- Initilisation

	make (a_count: INTEGER_32)
			-- Initialization of Current containing a_color_count colors
		require
			count_strictly_positive: a_count > 0
		local
			l_palette_pointer: POINTER
		do
			compare_objects
			l_palette_pointer := {GAME_SDL_EXTERNAL}.sdl_allocpalette (a_count)
			if l_palette_pointer.is_default_pointer then
				manage_error_pointer (l_palette_pointer, "Cannot create palette.")
			else
				make_shared (l_palette_pointer)
				shared := False
			end
		end

	make_shared (a_item: POINTER)
			-- Initialisation of Current using a_item as internal pointer
			-- Does not free memory pointer by a_item
		require
			item_not_null: not a_item.is_default_pointer
		do
			compare_objects
			make_by_pointer (a_item)
			shared := True
		end
	
feature -- Access

	count: INTEGER_32
			-- Number of item
		require
			palette_exist: exists
		do
			Result := {GAME_SDL_EXTERNAL}.get_sdl_palette_struct_ncolors (internal_pointer)
		end

	valid_index (i: INTEGER_32): BOOLEAN
			-- Is i a valid index?
		require else
			palette_exist: exists
		do
			Result := (i >= lower) and (i <= upper)
		end

	at alias "[]" (a_index: INTEGER_32): GAME_COLOR_READABLE assign put_i_th
			-- Retreive the color at a_index
		require else
			palette_exist: exists
		local
			l_color_pointer: POINTER
			l_color: GAME_COLOR
		do
			if exists then
				l_color_pointer := {GAME_SDL_EXTERNAL}.get_sdl_palette_struct_color_i (internal_pointer, a_index)
				create {GAME_SDL_COLOR} l_color.share_from_pointer (l_color_pointer)
			else
				create l_color.make_rgb (0, 0, 0)
			end
			Result := l_color
		end

	put_i_th (a_color: GAME_COLOR_READABLE; a_index: INTEGER_32)
			-- Put a_color at a_index-th position.
		local
			l_color: GAME_COLOR
		do
			if attached {GAME_COLOR} at (a_index) as la_color then
				la_color.red := a_color.red
				la_color.blue := a_color.blue
				la_color.green := a_color.green
				la_color.alpha := a_color.alpha
			end
		ensure
			is_set: create {GAME_COLOR}.make_from_other (at (a_index)) ~ a_color
		end

	index: INTEGER_32
			-- Index of current position

	start
			-- Move to first position if any.
		do
			index := 1
		end

	item: GAME_COLOR_READABLE
			-- Item at current position
		do
			Result := at (index)
		end

	is_empty: BOOLEAN
			-- Is there no element?
		do
			Result := False
		end

	forth
			-- Move to next position; if no next position,
			-- ensure that exhausted will be true.
		do
			index := index + 1
		end

	finish
			-- Move to last position.
		require else
			palette_exist: exists
		do
			if exists then
				index := count
			end
		end

	before: BOOLEAN
			-- Is there no valid position to the left of current one?
		do
			if not exists then
				Result := True
			else
				Result := index = 0
			end
		end

	after: BOOLEAN
			-- Is there no valid position to the right of current one?
		do
			if exists then
				Result := index = count + 1
			else
				Result := True
			end
		end

	back
			-- Move to previous position.
		do
			index := index - 1
		end

	is_equal (a_other: GAME_COLOR_PALETTE): BOOLEAN
			-- Is other attached to an object considered
			-- equal to current object?
		do
			from
				Result := count = a_other.count
				start
				a_other.start
			until
				not Result or off or a_other.off
			loop
				Result := Result and item ~ a_other.item
				forth
				a_other.forth
			end
		end
	
feature -- Measurement

	lower: INTEGER_32
			-- Minimum index.
		require else
			palette_exist: exists
		do
			Result := 1
		end

	upper: INTEGER_32
			-- Maximum index.
		require else
			palette_exist: exists
		do
			if exists then
				Result := count
			else
				Result := 1
			end
		end
	
feature {NONE} -- Implementation

	structure_size: INTEGER_32
			-- Size to allocate (in bytes).
		do
			Result := 0
		end

	shared: BOOLEAN
			-- Is Current share it's internal pointer with other object
			-- If so, it will not free the memory on dispose

	dispose
			-- Action to be executed just before garbage collection
			-- reclaims an object.
			-- Effect it in descendants to perform specific dispose
			-- actions. Those actions should only take care of freeing
			-- external resources; they should not perform remote calls
			-- on other objects since these may also be dead and reclaimed.
		do
			if not shared then
				{GAME_SDL_EXTERNAL}.sdl_freepalette (internal_pointer)
			end
		end
	
end -- class GAME_COLOR_PALETTE
